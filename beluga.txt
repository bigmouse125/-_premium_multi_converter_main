local function LoadBuildPreview(pBuildModel)

    -- Clear previous preview
    for _, child in ipairs(vu446:GetChildren()) do
        child:Destroy()
    end

    clone.Parent = vu446

    if not clone.PrimaryPart then
        for _, part in ipairs(clone:GetDescendants()) do
            if part:IsA("BasePart") then
                clone.PrimaryPart = part
                break
            end
        end
    end

    -- Scale and center preview
    local scale = Vector3.new(2, 2, 2)
    local centerPos = Vector3.new(0, 5, 0)
    for _, part in ipairs(clone:GetDescendants()) do
        if part:IsA("BasePart") then
            local offset = part.Position - clone:GetModelCFrame().p

            part.CFrame = CFrame.new(centerPos + offset)
        end
    end

    -- Move PrimaryPart to center
    if clone.PrimaryPart then
        clone:SetPrimaryPartCFrame(CFrame.new(centerPos))
    end
end
function ListBuilds(search)
    local files = listfiles("")
    local result = {}

    -- normalize search (optional: empty search shows all)
    search = search and search:lower() or ""

    for _, filename in next, files do
        local displayName = nil

        -- .build (lowercase)
        if filename:sub(-6) == ".build" then
            displayName = filename:sub(1, -7) .. " (build)"

        -- .Build (capital B)
        elseif filename:sub(-6) == ".Build" then
            displayName = filename:sub(1, -7) .. " (Build)"

        -- .obj
        elseif filename:sub(-4) == ".obj" then
            displayName = filename:sub(1, -5) .. " (obj)"

        -- .json
        elseif filename:sub(-5) == ".json" then
            displayName = filename:sub(1, -6) .. " (json)"
        end

        -- apply search filter
        if displayName then
            if search == "" or displayName:lower():find(search, 1, true) then
                table.insert(result, displayName)
            end
        end
    end

    return result
end
